    <script>
        document.getElementById('master-date').value = new Date().toISOString().slice(0, 16);

        let pipelineNodes = [];
        let currentType = '';

        // Çekmece İçi Dinamik Spesifikasyon Şablonları
        const templates = {
            cage: { title: "Cage Specifications", fields: `<div><label class="block text-[11px] font-medium text-slate-600 mb-1">Cage Code</label><input type="text" id="node-spec1" required placeholder="MAIN-CAGE-01" class="w-full px-3 py-1.5 border border-slate-200 rounded-lg text-xs"></div><div><label class="block text-[11px] font-medium text-slate-600 mb-1">Officer In Charge</label><input type="text" id="node-spec2" required placeholder="Ahmet Yilmaz" class="w-full px-3 py-1.5 border border-slate-200 rounded-lg text-xs"></div>` },
            chip: { title: "Chip Specifications", fields: `<div><label class="block text-[11px] font-medium text-slate-600 mb-1">Chip Set Type</label><input type="text" id="node-spec1" required placeholder="Premium Fiş" class="w-full px-3 py-1.5 border border-slate-200 rounded-lg text-xs"></div><div><label class="block text-[11px] font-medium text-slate-600 mb-1">Table Number</label><input type="text" id="node-spec2" required placeholder="Table VIP B" class="w-full px-3 py-1.5 border border-slate-200 rounded-lg text-xs"></div>` },
            petty: { title: "Petty Cash Specifications", fields: `<div><label class="block text-[11px] font-medium text-slate-600 mb-1">Expense Category</label><input type="text" id="node-spec1" required placeholder="Kurye Lojistik" class="w-full px-3 py-1.5 border border-slate-200 rounded-lg text-xs"></div><div><label class="block text-[11px] font-medium text-slate-600 mb-1">Invoice Reference</label><input type="text" id="node-spec2" required placeholder="INV-2026-X" class="w-full px-3 py-1.5 border border-slate-200 rounded-lg text-xs"></div>` },
            player: { title: "Player Specifications", fields: `<div><label class="block text-[11px] font-medium text-slate-600 mb-1">Player Account ID</label><input type="text" id="node-spec1" required placeholder="PLY-9821" class="w-full px-3 py-1.5 border border-slate-200 rounded-lg text-xs"></div><div><label class="block text-[11px] font-medium text-slate-600 mb-1">Full Name</label><input type="text" id="node-spec2" required placeholder="Jane Doe" class="w-full px-3 py-1.5 border border-slate-200 rounded-lg text-xs"></div>` }
        };

        // DRAWER'I AÇMA
        function openDrawer(type) {
            currentType = type;
            const config = templates[type];
            document.getElementById('drawer-title').textContent = config.title;
            document.getElementById('drawer-dynamic-fields').innerHTML = config.fields;
            
            // Drawer Formunu Sıfırla
            document.getElementById('node-amount').value = '';
            document.getElementById('drawerOverlay').classList.add('active');
        }

        // DRAWER'I KAPATMA
        function closeDrawer() {
            document.getElementById('drawerOverlay').classList.remove('active');
        }

        // DRAWER İÇİNDEKİ SEGMENETİ ANA LİSTEYE KAYDETME
        function saveNode(e) {
            e.preventDefault();
            
            const nodeData = {
                id: Date.now(),
                type: currentType,
                amount: parseFloat(document.getElementById('node-amount').value),
                currency: document.getElementById('node-currency').value,
                spec1: document.getElementById('node-spec1').value,
                spec2: document.getElementById('node-spec2').value
            };

            pipelineNodes.push(nodeData);
            renderPipelineList();
            closeDrawer();
        }

        // SEGMENETİ SİLME
        function removeNode(id) {
            pipelineNodes = pipelineNodes.filter(node => node.id !== id);
            renderPipelineList();
        }

        // ANA EKRANDAKİ ÖZET SATIRLARINI BASMA
        function renderPipelineList() {
            const listContainer = document.getElementById('inline-pipeline-list');
            const emptyMsg = document.getElementById('empty-state-msg');
            
            listContainer.innerHTML = '';

            if (pipelineNodes.length === 0) {
                listContainer.appendChild(emptyMsg);
                return;
            }

            pipelineNodes.forEach(node => {
                const card = document.createElement('div');
                card.className = "flex items-center justify-between p-3 bg-slate-50 rounded-xl border border-slate-200/60 hover:bg-slate-100/50 transition";
                card.innerHTML = `
                    <div class="flex items-center gap-4">
                        <!-- Küçük Sol İndikatör -->
                        <div class="w-1.5 h-8 bg-slate-700 rounded-full"></div>
                        <div class="flex flex-col">
                            <div class="flex items-center gap-2">
                                <span class="text-[10px] font-bold uppercase font-mono tracking-wider px-1.5 py-0.5 bg-slate-200 text-slate-700 rounded">${node.type}</span>
                                <span class="text-xs font-semibold text-slate-900">${node.amount.toLocaleString('tr-TR', {minimumFractionDigits:2})} ${node.currency}</span>
                            </div>
                            <span class="text-[11px] text-slate-400 mt-0.5">${node.spec1} • ${node.spec2}</span>
                        </div>
                    </div>
                    <!-- Silme Butonu -->
                    <button type="button" onclick="removeNode(${node.id})" class="text-slate-400 hover:text-red-500 p-2 rounded-lg transition cursor-pointer">
                        <i class="fa-solid fa-trash-can text-xs"></i>
                    </button>
                `;
                listContainer.appendChild(card);
            });
        }

        // ANA FORMU COMMIT ETME
        function submitMasterForm(e) {
            e.preventDefault();
            const masterRef = document.getElementById('master-ref').value;
            alert(`SaaS Multi-Step Commit Success!\nMaster Reference: ${masterRef}\nAttached Segments Total: ${pipelineNodes.length}`);
            console.log("Full Payload Map: ", pipelineNodes);
        }
    </script>
