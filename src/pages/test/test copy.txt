    <script>
        // Kurumsal Dinamik Alan Tanımlamaları
        const templates = {
            cage: { title: "Cage Specifications", fields: `<div><label class="block text-xs font-medium text-slate-600 mb-1">Cage Code</label><input type="text" id="spec-1" placeholder="MAIN-CAGE-01" required class="w-full px-3 py-2 border border-slate-200 rounded-xl text-sm bg-white"></div><div><label class="block text-xs font-medium text-slate-600 mb-1">Responsible Officer</label><input type="text" id="spec-2" placeholder="Ahmet Yılmaz" required class="w-full px-3 py-2 border border-slate-200 rounded-xl text-sm bg-white"></div>` },
            chip: { title: "Chip Specifications", fields: `<div><label class="block text-xs font-medium text-slate-600 mb-1">Chip Inventory Type</label><input type="text" id="spec-1" placeholder="Cash Chip" required class="w-full px-3 py-2 border border-slate-200 rounded-xl text-sm bg-white"></div><div><label class="block text-xs font-medium text-slate-600 mb-1">Table / VIP Lounge No</label><input type="text" id="spec-2" placeholder="Table 5" required class="w-full px-3 py-2 border border-slate-200 rounded-xl text-sm bg-white"></div>` },
            petty: { title: "Petty Cash Specifications", fields: `<div><label class="block text-xs font-medium text-slate-600 mb-1">Expense Category</label><input type="text" id="spec-1" placeholder="Office Supplies" required class="w-full px-3 py-2 border border-slate-200 rounded-xl text-sm bg-white"></div><div><label class="block text-xs font-medium text-slate-600 mb-1">Invoice / Receipt No</label><input type="text" id="spec-2" placeholder="INV-87123" required class="w-full px-3 py-2 border border-slate-200 rounded-xl text-sm bg-white"></div>` },
            player: { title: "Player Account Specifications", fields: `<div><label class="block text-xs font-medium text-slate-600 mb-1">Player Unique ID</label><input type="text" id="spec-1" placeholder="PLY-7712" required class="w-full px-3 py-2 border border-slate-200 rounded-xl text-sm bg-white"></div><div><label class="block text-xs font-medium text-slate-600 mb-1">Full Name</label><input type="text" id="spec-2" placeholder="John Doe" required class="w-full px-3 py-2 border border-slate-200 rounded-xl text-sm bg-white"></div>` }
        };

        let basket = [];
        const symbols = { TRY: '₺', USD: '$', EUR: '€' };

        function updateDynamicSection(type) {
            const data = templates[type];
            document.getElementById('dynamic-header').textContent = data.title;
            document.getElementById('dynamic-fields').innerHTML = data.fields;
        }

        document.querySelectorAll('input[name="tx_type"]').forEach(r => r.addEventListener('change', (e) => updateDynamicSection(e.target.value)));
        document.getElementById('input-currency').addEventListener('change', (e) => document.getElementById('currency-symbol').textContent = symbols[e.target.value]);

        // Tarih varsayılanı
        document.getElementById('input-date').value = new Date().toISOString().slice(0, 16);

        function addTransactionToBasket(e) {
            e.preventDefault();
            const selectedType = document.querySelector('input[name="tx_type"]:checked').value;
            
            const newItem = {
                id: Date.now(),
                type: selectedType,
                amount: parseFloat(document.getElementById('input-amount').value),
                currency: document.getElementById('input-currency').value,
                date: document.getElementById('input-date').value,
                payment: document.getElementById('input-payment').value,
                ref: document.getElementById('input-ref').value,
                spec1: document.getElementById('spec-1').value,
                spec2: document.getElementById('spec-2').value
            };

            basket.push(newItem);
            renderBasket();
            
            document.getElementById('input-amount').value = '';
            document.getElementById('input-ref').value = '';
            document.getElementById('spec-1').value = '';
            document.getElementById('spec-2').value = '';
        }

        function removeFromBasket(id) {
            basket = basket.filter(item => item.id !== id);
            renderBasket();
        }

        function renderBasket() {
            const listContainer = document.getElementById('basket-list');
            const emptyMsg = document.getElementById('empty-basket-msg');
            const countBadge = document.getElementById('basket-count');
            const totalLabel = document.getElementById('basket-total');
            const submitBtn = document.getElementById('btn-submit-all');

            listContainer.innerHTML = '';

            if (basket.length === 0) {
                listContainer.appendChild(emptyMsg);
                countBadge.textContent = "0 Items";
                totalLabel.textContent = "0.00 ₺";
                submitBtn.disabled = true;
                return;
            }

            submitBtn.disabled = false;
            countBadge.textContent = `${basket.length} ${basket.length > 1 ? 'Items' : 'Item'}`;

            let totalTRY = 0;

            basket.forEach(item => {
                totalTRY += item.amount; 

                const row = document.createElement('div');
                row.className = "flex items-center justify-between p-3 bg-slate-800/40 rounded-xl border border-slate-700/40 hover:bg-slate-800/80 transition group";
                row.innerHTML = `
                    <div class="flex flex-col gap-0.5">
                        <div class="flex items-center gap-2">
                            <span class="text-[9px] uppercase font-semibold px-1.5 py-0.5 rounded bg-slate-700 text-slate-300 font-mono tracking-wide">${item.type}</span>
                            <span class="font-medium text-slate-100 text-xs">${item.amount.toLocaleString('tr-TR', {minimumFractionDigits:2})} ${symbols[item.currency]}</span>
                        </div>
                        <span class="text-[10px] text-slate-400 font-mono">Ref: ${item.ref}</span>
                    </div>
                    <button onclick="removeFromBasket(${item.id})" type="button" class="text-slate-500 hover:text-red-400 p-1.5 rounded-lg hover:bg-slate-700/50 transition cursor-pointer">
                        <i class="fa-solid fa-trash-can text-xs"></i>
                    </button>
                `;
                listContainer.appendChild(row);
            });

            totalLabel.textContent = `${totalTRY.toLocaleString('tr-TR', {minimumFractionDigits:2})} ₺`;
        }

        function submitAllTransactions() {
            alert(`SaaS Pipeline: ${basket.length} adet işlem kuyruktan API'ye gönderildi.`);
            basket = [];
            renderBasket();
        }

        // Başlangıç
        updateDynamicSection('cage');
    </script>
