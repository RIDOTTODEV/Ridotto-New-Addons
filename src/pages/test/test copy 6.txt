        // YENİ ALT FORM KARTI EKLEME
        function appendSubForm(type) {
            formCounter++;
            const container = document.getElementById('sub-forms-container');
            const template = subFormTemplates[type];

            // Boş durum mesajını kaldır (varsa)
            const emptyMsg = document.getElementById('empty-state-msg');
            if (emptyMsg) emptyMsg.style.display = 'none';

            // Yeni Form Kartı Oluşturma
            const card = document.createElement('div');
            card.id = `sub-form-node-${formCounter}`;
            card.className = `border ${template.bgClass} rounded-2xl p-4 relative space-y-3 shadow-xs transition-all duration-200`;
            
            card.innerHTML = `
                <button type="button" onclick="removeSubForm(${formCounter})" class="absolute top-4 right-4 text-slate-400 hover:text-red-500 transition cursor-pointer p-1 rounded-lg hover:bg-slate-200/50">
                    <i class="fa-solid fa-xmark text-sm"></i>
                </button>

                <div class="flex items-center gap-2">
                    <span class="text-xs font-bold text-slate-800 tracking-tight">${template.title}</span>
                    <span class="text-[9px] font-mono bg-slate-200 text-slate-600 px-1.5 py-0.5 rounded uppercase">Node #${formCounter}</span>
                </div>

                <div class="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-4 gap-4">
                    ${template.fields}
                </div>
            `;

            container.appendChild(card);
        }

        // KART SİLME / CLEAR ETME FONKSİYONU
        function removeSubForm(id) {
            const card = document.getElementById(`sub-form-node-${id}`);
            if (card) {
                card.remove();
            }

            // Eğer hiç kart kalmadıysa boş durum mesajını geri getir
            const container = document.getElementById('sub-forms-container');
            if (container.children.length === 1 && container.firstElementChild.id === 'empty-state-msg') {
                document.getElementById('empty-state-msg').style.display = 'block';
            } else if (container.children.length === 0) {
                // Eğer DOM tamamen temizlendiyse şablonu baştan enjekte et
                container.innerHTML = `<div id="empty-state-msg" class="text-center py-8 border border-dashed border-slate-200 rounded-xl text-xs text-slate-400">No active transaction forms attached. Safe to commit as a standalone master entry or append sub-forms below.</div>`;
            }
        }

        // TÜM FORMLARI SUBMIT ETME
        function submitMasterForm(e) {
            e.preventDefault();
            const masterRef = document.getElementById('master-ref').value;
            const totalForms = document.getElementById('sub-forms-container').querySelectorAll('[id^="sub-form-node-"]').length;

            alert(`SaaS Commit Pipeline Successful!\nMaster Reference: ${masterRef}\nTotal Attached Forms Captured: ${totalForms}`);
        }
