        // YENİ TRANSACTİON BLOĞU OLUŞTURMA
        function createTransactionBlock(type) {
            blockCounter++;
            const listContainer = document.getElementById('transaction-pipeline-list');
            const data = subFormTemplates[type];

            const block = document.createElement('div');
            block.id = `tx-block-${blockCounter}`;
            block.className = "expanded border border-slate-200 rounded-xl overflow-hidden transition-all bg-slate-50/50";

            block.innerHTML = `
                <!-- Node Header (Collapse ve Silme) -->
                <div class="flex items-center justify-between px-4 py-2.5 bg-slate-100/70 border-b border-slate-200 select-none">
                    <div class="flex items-center gap-3 cursor-pointer flex-1" onclick="toggleCollapse(${blockCounter})">
                        <i class="fa-solid fa-chevron-right text-[10px] text-slate-400 transform transition-transform duration-200 rotate-90" id="arrow-${blockCounter}"></i>
                        <span class="text-xs font-semibold text-slate-700 tracking-wide">${data.label}</span>
                        <span class="text-[9px] uppercase tracking-wider font-mono bg-slate-200 text-slate-600 px-1.5 py-0.5 rounded">Node #${blockCounter}</span>
                    </div>
                    <button type="button" onclick="removeTransactionBlock(${blockCounter})" class="text-slate-400 hover:text-red-500 p-1 rounded transition cursor-pointer">
                        <i class="fa-solid fa-trash-can text-xs"></i>
                    </button>
                </div>

                <!-- Collapse İçerik Alanı -->
                <div class="collapse-content" id="content-${blockCounter}">
                    <div class="p-4 grid grid-cols-1 sm:grid-cols-2 md:grid-cols-4 gap-4 bg-white">
                        ${data.fields}
                    </div>
                </div>
            `;

            listContainer.appendChild(block);
            updateCounter();

            // Pürüzsüz animasyon için yükseklik hesaplama tetikleyici
            setTimeout(() => {
                const content = document.getElementById(`content-${blockCounter}`);
                content.style.maxHeight = content.scrollHeight + "px";
            }, 10);
        }

        // BLOK KALDIRMA
        function removeTransactionBlock(id) {
            const block = document.getElementById(`tx-block-${id}`);
            if (block) {
                block.remove();
                updateCounter();
            }
        }

        // COLLAPSE TETİKLEME
        function toggleCollapse(id) {
            const block = document.getElementById(`tx-block-${id}`);
            const content = document.getElementById(`content-${id}`);
            const arrow = document.getElementById(`arrow-${id}`);

            if (block.classList.contains('expanded')) {
                block.classList.remove('expanded');
                content.style.maxHeight = "0px";
                arrow.classList.remove('rotate-90');
            } else {
                block.classList.add('expanded');
                content.style.maxHeight = content.scrollHeight + "px";
                arrow.classList.add('rotate-90');
            }
        }

        // SAYAÇ GÜNCELLEME
        function updateCounter() {
            const count = document.getElementById('transaction-pipeline-list').children.length;
            document.getElementById('sub-tx-counter').textContent = `${count} Node${count > 1 ? 's' : ''} Linked`;
        }

        // FORM SUBMIT
        function submitForm(e) {
            e.preventDefault();
            const masterRef = document.getElementById('master-ref').value;
            const subCount = document.getElementById('transaction-pipeline-list').children.length;

            alert(`SaaS Commit Successful!\nMaster Ref: ${masterRef}\nTotal Sub-Transactions Linked: ${subCount}`);
        }
